\def\getangle(#1)(#2)#3{%
	\begingroup%
	\pgftransformreset%
	\pgfmathanglebetweenpoints{\pgfpointanchor{#1}{center}}{\pgfpointanchor{#2}{center}}%
	\expandafter\xdef\csname angle#3\endcsname{\pgfmathresult}%
	\endgroup%
}
